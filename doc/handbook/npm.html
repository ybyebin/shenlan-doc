    深蓝 2022-10-14

    # 查看 npm 的版本

    npm -v

    # 列出全局安装的模块 带上[--depth 0] 不深入到包的支点 更简洁

    npm list -g --depth 0

    # 安装检查升级模块

    npm install -g npm-check-updates

    # 查看所有 ncu 命令

    ncu -h

    # 检查当前可升级依赖

    ncu

    # 更新全部到最新版本

    ncu -a

    # 更新 package.json 文件

    ncu -u

    # linux 安装全局 npm 包,执行时报错找不到命令的问题

    echo -e "export PATH=$(npm prefix -g)/bin:$PATH" >> ~/.bashrc && source ~/.bashrc
